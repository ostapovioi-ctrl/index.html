    <div class="container">
        <h1>Привет из Mini App!</h1>
        <button id="sendBtn">Отправить привет боту</button>
        <div class="user-info" id="userInfo">
            Загрузка данных...
        </div>
    </div>

    <script>
        // Получаем объект Telegram WebApp
        const tg = window.Telegram.WebApp;
        
        // Обязательно: говорим Telegram, что приложение загружено
        tg.ready();
        
        // Раскрываем на весь экран
        tg.expand();

        // Показываем данные пользователя, если они есть
        const user = tg.initDataUnsafe?.user;
        const userInfoDiv = document.getElementById('userInfo');
        
        if (user) {
            userInfoDiv.innerHTML = `
                <strong>Ваши данные из Telegram:</strong><br>
                Имя: ${user.first_name} ${user.last_name || ''}<br>
                Username: ${user.username ? '@' + user.username : 'нет'}<br>
                ID: ${user.id}
            `;
        } else {
            userInfoDiv.innerHTML = 'Данные пользователя не получены (приложение открыто вне Telegram или нет данных)';
        }

        // Обработка кнопки отправки данных боту
        document.getElementById('sendBtn').addEventListener('click', () => {
            // Можно отправить любую строку боту
            tg.sendData("Привет от пользователя! Кнопка нажата.");
            // Показываем уведомление внутри Mini App (не обязательно)
            tg.showAlert("Сообщение отправлено боту!");
            // Закрыть приложение после отправки (раскомментируйте, если нужно)
            // tg.close();
        });

        // Дополнительно: показываем главную кнопку (например, "Закрыть")
        tg.MainButton.setText("Закрыть приложение");
        tg.MainButton.onClick(() => {
            tg.close();
        });
        tg.MainButton.show();
    </script>
